create proc dbo.pass_Login (
@username varchar(50),
@password varchar(50))
as
begin
select
count(*)
from dbo.UserLogin
WHERE username = @username
and password = hashbytes('sha2_512', @password + cast(salt as nvarchar(36)))
end
